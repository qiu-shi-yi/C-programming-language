## 穷举法

#### 简单实例1：鸡兔同笼问题

```c
#include <stdio.h>
#include <stdlib.h>

int main()
{
    int i = 98,k=320;
    int x=0; /*初始化鸡的个数*/
    int y;
    int a;
    do
    {
        x++;
        y = i-x;
        a = 2*x+4*y;
    }while(a!=k);
    printf("鸡的数量是 ：%d\t兔子的数量是：%d",x,y);
    return 0;
}

```
#### 简单实例2：韩信点兵问题

```c
#include <stdio.h>
#include <stdlib.h>

int main()
{
    int a,b,c,d,i = 0;
    do
    {
        i++;
        a = i%5;
        b = i%6;
        c = i%7;
        d = i%11;
    }while(a!=1||b!=5||c!=4||d!=5);
    printf("%d",i);
    return 0;
}
```
